    .ex-rows-fr {
        grid-template-rows: 50px 1fr 1fr; /* 58-1: Fixed height + equal remaining space */
    }

    /* 2. Grid Template Rows (Auto vs. Fraction) */
    .ex-rows-auto {
        grid-template-rows: auto 1fr; /* 58-2: Auto height for Row 1, 1fr for remaining Row 2 */
    }
    
    /* 3. Row Gap */
    .ex-row-gap {
        row-gap: 20px; /* 58-3: Space between rows */
    }
    
    /* 4. Column Gap */
    .ex-col-gap {
        column-gap: 10px; /* 58-4: Space between columns */
    }

    /* 5. Gap Shorthand */
    .ex-gap-shorthand {
        gap: 20px 5px; /* 58-5: 20px row gap, 5px column gap */
    }
