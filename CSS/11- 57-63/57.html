    }

    /* 2. Percentages and auto */
    .ex-percent-auto {
        grid-template-columns: 25% 50% auto; /* 57-2: Auto takes remaining space */
    }

    /* 3. Fractions (fr) */
    .ex-fraction {
        grid-template-columns: 3fr 1fr 1fr; /* 57-3: Space split into 5 parts (3/5, 1/5, 1/5) */
    }

    /* 4. Repeat Function */
    .ex-repeat {
        grid-template-columns: repeat(4, 1fr); /* 57-4: Creates 4 equal 1fr columns */
    }

    /* 5. Mixed Units */
    .ex-mixed {
        grid-template-columns: 100px 2fr auto 1fr; /* 57-5: Mix of fixed, flexible, and content-based sizing */
    }
