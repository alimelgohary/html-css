        align-items: stretch;
    }

    /* 2. Items at the Top */
    .ex-start {
        align-items: flex-start;
    }

    /* 3. Items at the Bottom */
    .ex-end {
        align-items: flex-end;
    }

    /* 4. Items Centered */
    .ex-center {
        align-items: center;
    }

    /* 5. Baseline Alignment */
    .ex-baseline {
        align-items: baseline;
    }

1. Align Items: stretch (Default)

Flex items stretch to fill the container along the cross axis, provided they don't have a fixed height set.
